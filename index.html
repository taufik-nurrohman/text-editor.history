Text Editor History 2.0.0

History extension for text editor.

Usage

Browser

<script src="./text-editor/index.min.js"></script>
<script src="./text-editor.history/index.min.js"></script>
<script>
Object.assign(TE.prototype, TE.History);
</script>

CommonJS

const TE = require('@taufik-nurrohman/text-editor');
const TE_History = require('@taufik-nurrohman/text-editor.history');

Object.assign(TE.prototype, TE_History);

ECMAScript

import TE from '@taufik-nurrohman/text-editor';
import * as TE_History from '@taufik-nurrohman/text-editor.history';

Object.assign(TE.prototype, TE_History);

Methods and Properties

editor.history(index)

Get history data.

editor.record(index)

Store current value and selection to history.

editor.loss(index)

Remove previous history data.

editor.undo()

Undo previous action.

editor.redo()

Redo previous action.
