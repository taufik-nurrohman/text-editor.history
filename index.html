Text Editor History 3.0.0

History extension for text editor.

Provides custom history feature to overcome the side effect of manipulating a <textarea> value that cannot automatically update the default history data.


      
Usage

      
Browser

      
<script src="./text-editor/index.min.js"></script>
<script src="./text-editor.history/index.min.js"></script>
<script>
  let editor = new TE(document.querySelector('textarea'));
</script>

      
CommonJS

      
const TE = require('@taufik-nurrohman/text-editor').default;
const {history} = require('@taufik-nurrohman/text-editor.history');

TE.state.with.push(history); // Load as a core extension

let editor = new TE(document.querySelector('textarea'), {
        with: [history] // Load as an optional extension
    });

      
ECMAScript

      
import TE from '@taufik-nurrohman/text-editor';
import {history} from '@taufik-nurrohman/text-editor.history';

TE.state.with.push(history); // Load as a core extension

let editor = new TE(document.querySelector('textarea'), {
        with: [history] // Load as an optional extension
    });

      
Methods

      
editor.history(index)

      
Get history data.

      
editor.loss(index)

      
Remove previous history data.

      
editor.record(index)

      
Store current value and selection to history.

      
editor.redo()

      
Redo previous action.

      
editor.undo()

      
Undo previous action.
